defined_text = {
	name = GetSVO_civilian_status_text
	text = {
		trigger = {
			check_variable = { THIS.SVO_Owner = 0 }
			NOT = { SVO = { has_country_flag = SVO_destroyed_citzenry } }
			OR = {
				is_core_of = SVO 
				SVO = { has_country_flag = SVO_can_give_non_cores }
			}
		}
		localization_key = SVO_civ_status_tt_available
	}
	text = {
		trigger = {
			OR = {
				check_variable = { THIS.SVO_Owner = -1 }
				SVO = { has_country_flag = SVO_destroyed_citzenry }
				check_variable = { THIS.SVO_Owner = 1 }
				AND = {
					NOT = { is_core_of = SVO }
					SVO = { NOT = { has_country_flag = SVO_can_give_non_cores } }
				}
			}
		}
		localization_key = SVO_civ_status_tt_unavailable
	}
}
